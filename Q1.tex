\section{Question 1}
\begin{enumerate}[label=(\alph*)]
    \item 
    We have,
    \[
    r = \sqrt{2 + 2} = 2,
    \]
    and by inspection, since $|x|=|y|$ and the complex number is in the fourth quadrant,
    \[
    \theta = -\pi/4.
    \]
    Therefore,
    \[
    \sqrt{2} - \sqrt{2}\,i = 2 e^{-i\pi/4}
    \]

    \item 
    Since, we found from (a) that $\sqrt{2} - \sqrt{2}\,i = 2 e^{-i\pi/4}$, then,
    \[
    \left(\sqrt{2} - \sqrt{2}\,i\right)^5 = 2^5 e^{-5i\pi/4}
    \]
    \[
    = 32 e^{-5i\pi/4}.
    \]

    \item 
    We calculate,
    \[
    1 + \sqrt{3}\,i = \sqrt{1 + 3}\,\exp\left({i\arctan(\sqrt{3}/1)}\right)
    \]
    \[
    = 2e^{i\pi/3},
    \]
    which is valid since $1 + \sqrt{3}\,i$ is in the first quadrant.
    
    Now since we know the numerator from (a),
    \[
    \frac{\sqrt{2}-\sqrt{2}\,i}{1 + \sqrt{3}\,i} = \frac{2 e^{-i\pi/4}}{2e^{i\pi/3}}
    \]
    \[
    = e^{-i\pi/4 - i\pi/3}
    \]
    \[
    = e^{-i\pi(1/4 + 1/3)}
    \]
    \[
    = e^{-i(7\pi/12)}.
    \]

    \item 
    \[
    (1 - i)^{-1} = \exp\left(-i\,\mathrm{Log}(1-i)\right)
    \]
    \[
    = \exp\left(-i\,(\log|\sqrt{2}| - i\pi/4 )\right)
    \]
    \[
    = e^{-\pi/4}e^{i\log\sqrt{2}}.
    \]
\end{enumerate}
