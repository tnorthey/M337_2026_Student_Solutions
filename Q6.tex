    \item 
    By HB D2 2.1 p89, the iteration sequence
    \begin{align*}
        z_{n+1} &= (z_n + 3i)(iz_n + 1), \qquad n = 0,1,2, \ldots \\
        &= iz^2_n - 2z_n + 3i
    \end{align*}
    is conjugate to the iteration sequence
\[ w_{n+1} = w^2_n + d  \qquad n= 0,1,2, \ldots, \]
where
\[ d = -3 - \frac{2}{2} - \frac{1}{4}(2^2) = -3-1-1 = -5 . \]
The conjugating function is $ h(z) = iz - \dfrac{2}{2} = iz - 1 $. \href{}{}
\vs \\
Thus, $ w_0 = h(z_0) = i(i)-1 = -2 $, as required.
\vvs \\

\begin{enumerate}[label=(\roman*)]
    \item 
    By the diagram in HB D2 4.4 p91, $ c = -\dfrac{6}{5} + \frac{1}{10}i $, looks like it is in the Mandelbrot set.
    \vs \\
    Now,
    \begin{align*}
        |c+1| &=\left| \frac{1}{5} + \frac{1}{10}i  \right| \\ 
              &= \sqrt{ \frac{1}{25} + \frac{1}{100} } \\
              &= \sqrt{ \frac{5}{100} } \\
              &= \sqrt{\frac{1}{20}} < \sqrt{ \frac{1}{16} } = \frac{1}{4}.
    \end{align*}
    So, by HB D2 4.11(b), $P_c$ has an attracting 2-cycle. 
    \vs \\
    Hence, by HB D2 4.10, p92, $ -\dfrac{6}{5} + \dfrac{1}{10}i $ is in the Mandelbrot set.
    \vvs \\
    \item 
    As $ c = 1 + \dfrac{2}{5}i $,
    \begin{align*}
        P_c(0)   &= 1 + \frac{2}{5}i \\
        P^2_c(0) &= \frac{1}{5} + \frac{4}{5}i + 1 +  \frac{2}{5}i = \frac{6}{5}(1+i) \\
        P^3_c(0) &= \frac{36}{25}(2i) + 1 + \frac{2}{5}i \\
                 &= 1 + \frac{82}{25}i.
    \end{align*}
    But $ | P^3_c(0) | > \left| \dfrac{82}{25}i \right| > 3 >2 $.
    \vs \\
    So, by HB D2 4.6 p92, $ c = 1 + \dfrac{2}{5}i $ is not in the Mandelbrot set.
\end{enumerate}
\end{enumerate}
